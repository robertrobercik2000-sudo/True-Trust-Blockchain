1//! KMAC256 cryptographic functions for wallet operations
2//! Uses SHAKE256 (SHA3 XOF) as the underlying primitive
3
4use sha3::{Shake256, digest::{Update, ExtendableOutput, XofReader}};
5
6/// KMAC256 key derivation function
7/// Derives a 32-byte key from password, label, and salt
8pub fn kmac256_derive_key(password: &[u8], label: &[u8], salt: &[u8]) -> [u8; 32] {
9    let mut hasher = Shake256::default();
10    Update::update(&mut hasher, b"KMAC256-KDF-v1");
11    Update::update(&mut hasher, &(password.len() as u64).to_le_bytes());
12    Update::update(&mut hasher, password);
13    Update::update(&mut hasher, &(label.len() as u64).to_le_bytes());
14    Update::update(&mut hasher, label);
15    Update::update(&mut hasher, &(salt.len() as u64).to_le_bytes());
16    Update::update(&mut hasher, salt);
17    
18    let mut reader = hasher.finalize_xof();
19    let mut out = [0u8; 32];
20    XofReader::read(&mut reader, &mut out);
21    out
22}

24/// KMAC256 XOF (extendable output function)
25/// Generates a variable-length output stream
26pub fn kmac256_xof(input: &[u8], label: &[u8], salt: &[u8], output_len: usize) -> Vec<u8> {
27    let mut hasher = Shake256::default();
28    Update::update(&mut hasher, b"KMAC256-XOF-v1");
29    Update::update(&mut hasher, &(input.len() as u64).to_le_bytes());
30    Update::update(&mut hasher, input);
31    Update::update(&mut hasher, &(label.len() as u64).to_le_bytes());
32    Update::update(&mut hasher, label);
33    Update::update(&mut hasher, &(salt.len() as u64).to_le_bytes());
34    Update::update(&mut hasher, salt);
35    
36    let mut reader = hasher.finalize_xof();
37    let mut out = vec![0u8; output_len];
38    XofReader::read(&mut reader, &mut out);
39    out
40}

42/// KMAC256 tag generation (MAC)
43/// Generates a 32-byte authentication tag

52    Update::update(&mut hasher, message);
53    
54    let mut reader = hasher.finalize_xof();

60#[cfg(test)]
61mod tests {
62    use super::*;
63
64    #[test]
65    fn test_kmac256_derive_key_deterministic() {
66        let k1 = kmac256_derive_key(b"password", b"label", b"salt");
67        let k2 = kmac256_derive_key(b"password", b"label", b"salt");
68        assert_eq!(k1, k2);
69    }
70
71    #[test]
72    fn test_kmac256_xof_variable_length() {
73        let out1 = kmac256_xof(b"input", b"label", b"salt", 16);
74        let out2 = kmac256_xof(b"input", b"label", b"salt", 32);
75        assert_eq!(out1.len(), 16);
76        assert_eq!(out2.len(), 32);
77        assert_eq!(&out2[..16], &out1[..]);
78    }
79
80    #[test]
81    fn test_kmac256_tag_deterministic() {
82        let t1 = kmac256_tag(b"key", b"label", b"message");
