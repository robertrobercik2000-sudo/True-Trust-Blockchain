1//! KMAC256 hash functions for consensus operations
2//! Uses SHAKE256 (SHA3 XOF) as the underlying primitive
3
4use sha3::{Shake256, digest::{Update, ExtendableOutput, XofReader}};
5
6/// KMAC256 hash (32 bytes output) - deterministic hash function
7/// Uses a fixed key for consensus operations (domain separation via label)
8pub fn kmac256_hash(label: &[u8], inputs: &[&[u8]]) -> [u8; 32] {
9    // Fixed key for consensus operations (domain separation via label)
10    const CONSENSUS_KEY: &[u8] = b"TT-CONSENSUS-KMAC256";
11    
12    let mut hasher = Shake256::default();
13    Update::update(&mut hasher, b"KMAC256-HASH-v1");
14    Update::update(&mut hasher, &(CONSENSUS_KEY.len() as u64).to_le_bytes());
15    Update::update(&mut hasher, CONSENSUS_KEY);
16    Update::update(&mut hasher, &(label.len() as u64).to_le_bytes());
17    Update::update(&mut hasher, label);
18    for input in inputs {
19        Update::update(&mut hasher, &(input.len() as u64).to_le_bytes());
20        Update::update(&mut hasher, input);
21    }
22    
23    let mut reader = hasher.finalize_xof();
24    let mut out = [0u8; 32];
25    XofReader::read(&mut reader, &mut out);
26    out
27}
